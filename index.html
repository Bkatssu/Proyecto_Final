MRU

Movimiento Rectilíneo Uniforme

Velocidad constante. La aceleración es cero. La posición varía linealmente con el tiempo.

x = x₀ + v · t
a = 0
v = constante
Gráfica x(t) → línea recta
MRUV

Movimiento Rectilíneo Uniformemente Variado

Aceleración constante distinta de cero. La velocidad cambia de forma lineal.

x = x₀ + v₀t + ½at²
v = v₀ + at
v² = v₀² + 2a(x − x₀)
a = constante ≠ 0
Gráfica v(t) → línea recta
Gráfica x(t) → parábola
CAÍDA LIBRE

Caída Libre

MRUV especial con a = g = 9.81 m/s² hacia abajo. Sin resistencia del aire.

y = y₀ + v₀t − ½gt²
v = v₀ − gt
g = 9.81 m/s²
a = −g = −9.81 m/s²
Eje vertical
Vacío o sin fricción
CÁLCULO DIFERENCIAL

Velocidad y Aceleración Instantáneas

Las variables cinemáticas son derivadas sucesivas de la posición respecto al tiempo.

Posición

v = dx/dt
→ d/dt
a = dv/dt
En discreta: v ≈ Δx/Δt
En discreta: a ≈ Δv/Δt

class MotorCinematico {
  constructor(x0, v0, a) {
    this.x0 = x0;   // Posición inicial (m)
    this.v0 = v0;   // Velocidad inicial (m/s)
    this.a  = a;    // Aceleración (m/s²)
  }

  // Posición en el tiempo t
  posicion(t) {
    // MRUV: x = x0 + v0*t + (1/2)*a*t^2
    return this.x0 + this.v0 * t + 0.5 * this.a * t * t;
  }

  // Velocidad en el tiempo t
  velocidad(t) {
    // v = v0 + a*t
    return this.v0 + this.a * t;
  }

  // Aceleración (constante en cinemática básica)
  aceleracion() {
    return this.a;
  }

  // Clasifica el tipo de movimiento
  clasificar() {
    if (Math.abs(this.a) < 0.05) return "MRU";
    if (Math.abs(this.a + 9.81) < 0.1) return "Caída libre";
    return "MRUV";
  }

  // Distancia recorrida entre t1 y t2
  distancia(t1, t2) {
    return Math.abs(this.posicion(t2) - this.posicion(t1));
  }
}

// Uso:
const motor = new MotorCinematico(0, 5, 2);
console.log(motor.clasificar());    // "MRUV"
console.log(motor.posicion(3));     // 0 + 5*3 + 0.5*2*9 = 24
console.log(motor.velocidad(3));    // 5 + 2*3 = 11

"""

import cv2
import numpy as np
import matplotlib.pyplot as plt

# ──────────────────────────────────────
# MOTOR DE CINEMÁTICA
# ──────────────────────────────────────
class MotorCinematico:
    def __init__(self, escala_px_m=100):
        """escala_px_m: cuántos píxeles equivalen a 1 metro"""
        self.escala = escala_px_m
        self.posiciones = []   # metros
        self.velocidades = []  # m/s
        self.aceleraciones = [] # m/s²
        self.tiempos = []      # segundos
        self.prev_pos = None
        self.prev_vel = None
        self.prev_t   = None

    def registrar(self, px, py, t):
        """Registra posición en píxeles y calcula variables físicas"""
        x = px / self.escala  # convertir a metros
        self.tiempos.append(t)
        self.posiciones.append(x)

        if self.prev_pos is not None and self.prev_t is not None:
            dt = t - self.prev_t
            if dt > 0:
                v = (x - self.prev_pos) / dt
            else:
                v = 0
        else:
            v = 0

        self.velocidades.append(v)

        if self.prev_vel is not None and self.prev_t is not None:
            dt = t - self.prev_t
            a = (v - self.prev_vel) / dt if dt > 0 else 0
        else:
            a = 0

        self.aceleraciones.append(a)
        self.prev_pos = x
        self.prev_vel = v
        self.prev_t   = t

    def clasificar(self):
        if len(self.aceleraciones) < 5:
            return "Sin datos"
        a_media = np.mean(self.aceleraciones[-10:])
        if abs(a_media) < 0.2:
            return "MRU"
        elif abs(a_media + 9.81) < 1.0:
            return "Caída libre"
        else:
            return "MRUV"


# ──────────────────────────────────────
# DETECCIÓN DE OBJETO (COLOR)
# ──────────────────────────────────────
def detectar_objeto(frame, color_hsv_min, color_hsv_max):
    """Detecta objeto por color usando HSV. Retorna centroide (cx, cy)."""
    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
    mascara = cv2.inRange(hsv, color_hsv_min, color_hsv_max)
    mascara = cv2.GaussianBlur(mascara, (5, 5), 0)
    contornos, _ = cv2.findContours(mascara, cv2.RETR_EXTERNAL,
                                    cv2.CHAIN_APPROX_SIMPLE)
    if contornos:
        mayor = max(contornos, key=cv2.contourArea)
        if cv2.contourArea(mayor) > 200:
            M = cv2.moments(mayor)
            if M["m00"] > 0:
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                return cx, cy
    return None, None


# ──────────────────────────────────────
# LOOP PRINCIPAL
# ──────────────────────────────────────
def main(fuente=0):
    """
    fuente: 0 para webcam, o ruta a video 'video.mp4'
    Objeto a detectar: pelota naranja (ajustar rangos HSV)
    """
    cap = cv2.VideoCapture(fuente)
    motor = MotorCinematico(escala_px_m=100)

    # Rango HSV para objeto naranja
    hsv_min = np.array([5,  120, 120])
    hsv_max = np.array([25, 255, 255])

    t_inicio = None

    while True:
        ret, frame = cap.read()
        if not ret:
            break

        import time
        ahora = time.time()
        if t_inicio is None:
            t_inicio = ahora
        t = ahora - t_inicio

        cx, cy = detectar_objeto(frame, hsv_min, hsv_max)

        if cx is not None:
            motor.registrar(cx, cy, t)

            # Dibujar en pantalla
            cv2.circle(frame, (cx, cy), 20, (0, 200, 100), 3)
            tipo = motor.clasificar()
            pos = motor.posiciones[-1] if motor.posiciones else 0
            vel = motor.velocidades[-1] if motor.velocidades else 0
            acc = motor.aceleraciones[-1] if motor.aceleraciones else 0

            cv2.putText(frame, f"Tipo: {tipo}",       (10, 30),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)
            cv2.putText(frame, f"x = {pos:.2f} m",    (10, 60),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)
            cv2.putText(frame, f"v = {vel:.2f} m/s",  (10, 90),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)
            cv2.putText(frame, f"a = {acc:.2f} m/s²", (10,120),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)
            cv2.putText(frame, f"t = {t:.1f} s",      (10,150),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255,255,0), 2)

        cv2.imshow("Monitor de Movimiento — Fisica 1", frame)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()

    # ── Graficar resultados finales ──
    if motor.tiempos:
        fig, axs = plt.subplots(3, 1, figsize=(10, 8),
                                facecolor='#0f0f0f')
        fig.suptitle("Análisis Cinemático — Física 1", color='white',
                     fontsize=14)

        for ax in axs:
            ax.set_facecolor('#1a1a1a')
            ax.tick_params(colors='white')
            ax.xaxis.label.set_color('white')
            ax.yaxis.label.set_color('white')
            for spine in ax.spines.values():
                spine.set_edgecolor('#444')

        axs[0].plot(motor.tiempos, motor.posiciones, '#5DCAA5', lw=2)
        axs[0].set_ylabel("Posición (m)")
        axs[0].set_title(f"Tipo detectado: {motor.clasificar()}",
                         color='#5DCAA5', fontsize=11)

        axs[1].plot(motor.tiempos, motor.velocidades, '#85B7EB', lw=2)
        axs[1].set_ylabel("Velocidad (m/s)")

        axs[2].plot(motor.tiempos, motor.aceleraciones, '#EF9F27', lw=2)
        axs[2].set_ylabel("Aceleración (m/s²)")
        axs[2].set_xlabel("Tiempo (s)")

        plt.tight_layout()
        plt.savefig("resultados_movimiento.png", dpi=150,
                    bbox_inches='tight', facecolor='#0f0f0f')
        plt.show()
        print("Gráfica guardada como resultados_movimiento.png")


if __name__ == "__main__":
    main(fuente=0)   # 0 = webcam, o pon 'video.mp4'
algoritmo — Pseudocódigo del sistema 
ALGORITMO: Sistema de Monitoreo de Movimiento
================================================
ENTRADAS: x0, v0, a, t_total
SALIDAS:  x(t), v(t), a(t), tipo_movimiento

PASO 1 — INICIALIZACIÓN
  t ← 0
  Δt ← 0.05 s  (paso de integración)
  x_prev ← x0
  v_prev ← v0

PASO 2 — CLASIFICAR MOVIMIENTO
  SI |a| < 0.05 ENTONCES
    tipo ← "MRU"
  SINO SI |a + 9.81| < 0.1 ENTONCES
    tipo ← "Caída libre"
  SINO
    tipo ← "MRUV"
  FIN SI

PASO 3 — BUCLE DE SIMULACIÓN
  MIENTRAS t ≤ t_total HACER

    // Calcular con ecuaciones de cinemática
    SI tipo = "MRU" ENTONCES
      x ← x0 + v0 * t
      v ← v0
      a_inst ← 0

    SINO  // MRUV o Caída libre
      x ← x0 + v0*t + 0.5*a*t²
      v ← v0 + a*t
      a_inst ← a
    FIN SI

    // Calcular con diferencias finitas (detección real)
    v_discreta ← (x - x_prev) / Δt
    a_discreta ← (v_discreta - v_prev) / Δt

    // Registrar datos
    GUARDAR (t, x, v, a_inst)
    ACTUALIZAR gráficas x(t), v(t), a(t)
    MOSTRAR métricas en pantalla

    // Avanzar
    x_prev ← x
    v_prev ← v_discreta
    t ← t + Δt

  FIN MIENTRAS

PASO 4 — VALIDACIÓN
  PARA cada t en [0, 0.5, 1.0, 1.5, 2.0]:
    x_teorico ← x0 + v0*t + 0.5*a*t²
    v_teorico ← v0 + a*t
    error% ← |x_medido - x_teorico| / |x_teorico| * 100
    IMPRIMIR t, x_teorico, x_medido, error%
  FIN PARA
